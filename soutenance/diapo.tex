\section{Le projet}
	\subsection{Présentation du sujet}
	\begin{frame}{Le projet : Application Android}
		\begin{itemize}
			\item Cueillette de pommes avec le doigt
			\item Ramasser le plus de pommes possibles
			\item Possibilité de régler certains paramètres
		\end{itemize}

	\end{frame}
	\subsection{Principales fonctionnalités}
	\begin{frame}{Principales fonctionnalités développées}
		\begin{block}{Gameplay}
			\begin{itemize}
				\item Tombée des fruits
				\item Accélération progressive des pommes
				\item Calcul de score
				\item Disparition des fruits une fois cueillis
				\item Mémorisation du meilleur score
			\end{itemize}
		\end{block}
		\pause
		\begin{block}{Paramétrage}
			\begin{itemize}
				\item Changement de la vitesse initiale
				\item Changement du nombre de fruits
				\item Changement du nombre de vies 
			\end{itemize}
		\end{block}
	\end{frame}
